后台管理系统角色管理和用户权限设计方案
项目背景
本项目为一个后台管理系统，使用 Nuxt 4.x 集成前端页面和后端服务，基于 Supabase 的 PostgreSQL 数据库和内置认证系统（auth.users），启用 Row Level Security (RLS) 确保数据安全。目前，菜单表（menus）和相关接口已开发并联调完成。接下来需要设计和实现角色表（roles）、角色管理页面及相关接口，并规划用户管理和权限获取流程。用户数据存储在 auth.users 表，手机号和部门 ID 存储在 raw_user_meta_data 中，无需额外 users 表。后续将创建部门表（departments）和部门管理页面。
当前进展

菜单表（menus）：已完成表设计，包含字段：id, name, icon, path, parent_id, sort, status, permission, type, created_at, updated_at。
菜单接口：已实现并联调完成，包括：
GET /api/menus: 获取菜单列表（支持树形结构）。
GET /api/menus/:id: 获取单个菜单详情。
POST /api/menus: 添加菜单。
PUT /api/menus/:id: 更新菜单。
DELETE /api/menus/:id: 删除菜单。
DELETE /api/menus/batch: 批量删除菜单。
POST /api/menus/sort: 调整菜单排序。

技术栈：Nuxt 4.x（前端 + 后端服务），Supabase（PostgreSQL + 认证系统），TypeScript，Supabase CLI（生成类型）。

接下来计划：角色管理和用户权限
一、用户管理方案

使用 auth.users：利用 Supabase 内置的 auth.users 表管理用户，存储核心信息（如 id (UUID)、email、password（加密）、created_at、updated_at）。
自定义字段：在 auth.users.raw_user_meta_data (JSONB) 中存储：
phone: 用户手机号。
department_id: 部门 ID，关联后续的 departments 表。
可选字段：如 nickname（昵称）。

示例 raw_user_meta_data：{
"phone": "1234567890",
"department_id": "dept_001",
"nickname": "User Name"
}

理由：
避免额外 users 表，简化架构。
raw_user_meta_data 灵活支持自定义字段，适合存储简单属性。
Supabase 认证系统自动处理密码加密、Token 管理等。

二、角色表设计（roles）
角色表用于存储系统角色信息，与用户和菜单权限关联，支持多对多关系。id 字段为主键（BIGINT），由 Supabase 或后端逻辑处理生成。
表结构
CREATE TABLE roles (
id BIGINT PRIMARY KEY,
name VARCHAR(50) NOT NULL,
code VARCHAR(50) NOT NULL UNIQUE,
description VARCHAR(255),
status SMALLINT NOT NULL DEFAULT 1 CHECK (status IN (0, 1)),
created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
COMMENT ON TABLE roles IS '角色表';
COMMENT ON COLUMN roles.id IS '角色 ID，唯一标识角色';
COMMENT ON COLUMN roles.name IS '角色名称';
COMMENT ON COLUMN roles.code IS '角色编码，唯一标识，如 admin, user';
COMMENT ON COLUMN roles.description IS '角色描述';
COMMENT ON COLUMN roles.status IS '状态（0：禁用，1：启用）';
COMMENT ON COLUMN roles.created_at IS '创建时间';
COMMENT ON COLUMN roles.updated_at IS '更新时间';

-- 启用 RLS
ALTER TABLE roles ENABLE ROW LEVEL SECURITY;
-- 允许认证用户读取
CREATE POLICY "Allow authenticated read on roles" ON roles
FOR SELECT USING (auth.role() = 'authenticated');
-- 允许管理员用户增删改
CREATE POLICY "Allow admin full access on roles" ON roles
FOR ALL USING (auth.uid() IN (SELECT user_id FROM users_role WHERE role_id = (SELECT id FROM roles WHERE code = 'admin')));

字段说明

id: 主键（BIGINT），唯一标识角色，由 Supabase 或后端逻辑生成（如使用 UUID 或其他算法）。
name: 角色名称，如“管理员”“普通用户”。
code: 角色编码，唯一标识（如 admin），便于程序逻辑判断。
description: 可选描述。
status: 0（禁用）或 1（启用）。
created_at / updated_at: 时间戳，支持时区。
RLS: 认证用户可读取，管理员可完全操作。

索引

主键索引：id（自动创建）。
唯一索引：code。

（2）获取用户菜单权限 (server/api/auth/menus.get.ts)
import { createClient } from '@supabase/supabase-js';
import { Database } from '~/types/database.types';
import jwt from 'jsonwebtoken';

const supabase = createClient<Database>(process.env.SUPABASE_URL!, process.env.SUPABASE_ANON_KEY!);

export default defineEventHandler(async (event) => {
  const token = getHeader(event, 'Authorization')?.replace('Bearer ', '');
  if (!token) throw createError({ statusCode: 401, message: '未提供Token' });

  let decoded: { userId: string; roles: string[] };
  try {
    decoded = jwt.verify(token, process.env.JWT_SECRET!) as { userId: string; roles: string[] };
  } catch (error) {
    throw createError({ statusCode: 401, message: '无效的Token' });
  }

  const { data: menus, error } = await supabase
    .from('menus')
    .select('*')
    .in(
      'id',
      supabase
        .from('roles_menu')
        .select('menu_id')
        .in(
          'role_id',
          supabase
            .from('users_role')
            .select('role_id')
            .eq('user_id', decoded.userId)
        )
    )
    .eq('status', 1)
    .order('sort', { ascending: true })
    .order('id', { ascending: true });

  if (error) throw createError({ statusCode: 500, message: '获取菜单失败：' + error.message });

  const buildMenuTree = (menus: Database['public']['Tables']['menus']['Row'][], parentId: number = 0) => {
    return menus
      .filter((menu) => menu.parent_id === parentId)
      .map((menu) => ({ ...menu, children: buildMenuTree(menus, menu.id) }));
  };

  return { code: 0, message: 'success', data: buildMenuTree(menus) };
});

import { createClient } from '@supabase/supabase-js';
import { Database } from '~/types/database.types';

const supabase = createClient<Database>(process.env.SUPABASE_URL!, process.env.SUPABASE_ANON_KEY!);

export default defineEventHandler(async (event) => {
  const query = getQuery(event);
  const keyword = query.keyword as string | undefined;
  const status = query.status ? parseInt(query.status as string) : undefined;
  const page = parseInt(query.page as string) || 1;
  const pageSize = parseInt(query.pageSize as string) || 10;

  let queryBuilder = supabase
    .from('roles')
    .select('*', { count: 'exact' })
    .order('id', { ascending: true })
    .range((page - 1) * pageSize, page * pageSize - 1);

  if (keyword) {
    queryBuilder = queryBuilder.ilike('name', `%${keyword}%`).or(`code.ilike.%${keyword}%`);
  }
  if (status !== undefined) {
    queryBuilder = queryBuilder.eq('status', status);
  }

  const { data, error, count } = await queryBuilder;
  if (error) throw createError({ statusCode: 500, message: error.message });

  return {
    code: 0,
    message: 'success',
    data,
    total: count
  };
});

import { createClient } from '@supabase/supabase-js';
import { Database } from '~/types/database.types';

const supabase = createClient<Database>(process.env.SUPABASE_URL!, process.env.SUPABASE_ANON_KEY!);

export default defineEventHandler(async (event) => {
  const roleId = parseInt(event.context.params.id);
  const { menuIds } = await readBody(event);

  const { error: deleteError } = await supabase
    .from('roles_menu')
    .delete()
    .eq('role_id', roleId);
  if (deleteError) throw createError({ statusCode: 500, message: deleteError.message });

  const inserts = menuIds.map((menuId: number) => ({ role_id: roleId, menu_id: menuId }));
  const { error: insertError } = await supabase
    .from('roles_menu')
    .insert(inserts);
  if (insertError) throw createError({ statusCode: 500, message: insertError.message });

  return { code: 0, message: '权限分配成功' };
});

Validate raw_user_meta_data.department_id against departments:CREATE FUNCTION validate_department_id() RETURNS TRIGGER AS $$
BEGIN
  IF NEW.raw_user_meta_data->>'department_id' IS NOT NULL THEN
    IF NOT EXISTS (SELECT 1 FROM departments WHERE id = NEW.raw_user_meta_data->>'department_id') THEN
      RAISE EXCEPTION 'Invalid department_id';
    END IF;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

BEFORE INSERT OR UPDATE ON auth.users
